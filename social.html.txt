// Настройка Firebase (Твой проект)
const firebaseConfig = { apiKey: "AIzaSyDw7KtHos2d88AdeK5eTQSqd5TaAMl2uwc", projectId: "e-my-fam" };
firebase.initializeApp(firebaseConfig);
const db = firebase.firestore();

// Дефолтная аватарка, если своей нет
const defaultAvatar = "https://i.imgur.com/6VBx3io.png";

// ЗАГРУЗКА И СОХРАНЕНИЕ ПРОФИЛЯ
function loadProfile() {
    const name = localStorage.getItem('social_nik') || "";
    const avatar = localStorage.getItem('social_avatar') || defaultAvatar;
    
    document.getElementById('userName').value = name;
    document.getElementById('userAvatarUrl').value = localStorage.getItem('social_avatar') || ""; // Показываем пустую строку в инпуте, если не задано
    
    // Обновляем аватарки на странице
    document.getElementById('myAvatarDisplay').src = avatar;
    document.querySelectorAll('.my-avatar-sync').forEach(img => img.src = avatar);
}

function saveProfile() {
    const name = document.getElementById('userName').value;
    const avatarUrl = document.getElementById('userAvatarUrl').value;
    
    localStorage.setItem('social_nik', name);
    // Если поле пустое, сохраняем дефолтную
    localStorage.setItem('social_avatar', avatarUrl.trim() || defaultAvatar);
    
    loadProfile(); // Перезагрузить, чтобы обновить картинки
    alert("Профиль обновлен!");
}

// Запускаем загрузку профиля при старте
loadProfile();


// РАБОТА С ЛЕНТОЙ (ПОСТЫ С ФОТО)
function sendPost() {
    const text = document.getElementById('postText').value;
    const imgUrl = document.getElementById('postImgUrl').value;
    const name = localStorage.getItem('social_nik') || "Аноним";
    const avatar = localStorage.getItem('social_avatar') || defaultAvatar;
    
    if(text.trim() || imgUrl.trim()) {
        db.collection("posts").add({
            u: name,
            ava: avatar, // Сохраняем аватарку В ПОСТ, чтобы она не менялась потом
            t: text,
            img: imgUrl.trim(), // Ссылка на фото поста
            likes: 0,
            likedBy: [], // Список тех, кто лайкнул (чтобы не лайкали дважды)
            time: Date.now()
        });
        // Очистить поля
        document.getElementById('postText').value = "";
        document.getElementById('postImgUrl').value = "";
    } else {
        alert("Напиши что-нибудь или вставь ссылку на фото!");
    }
}

// Слушаем изменения в ленте
db.collection("posts").orderBy("time", "desc").onSnapshot(s => {
    const feed = document.getElementById('feed');
    feed.innerHTML = "";
    
    if(s.empty) {
        feed.innerHTML = `<p style="text-align:center; color:#555; padding:20px;">Лента пуста. Будь первым!</p>`;
        return;
    }
    
    s.forEach(doc => {
        const d = doc.data();
        const postId = doc.id;
        // Проверяем, лайкнул ли я этот пост (хранится в массиве likedBy)
        const isLiked = d.likedBy && d.likedBy.includes(localStorage.getItem('social_nik'));

        feed.innerHTML += `
            <div class="post">
                <div class="post-header">
                    <img src="${d.ava || defaultAvatar}" class="post-avatar" alt="Ava">
                    <div class="post-info">
                        <b>${d.u}</b><br>
                        <small>${new Date(d.time).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</small>
                    </div>
                </div>
                <div class="post-content">
                    ${d.t ? `<div class="post-text">${d.t}</div>` : ''}
                    ${d.img ? `<img src="${d.img}" alt="Post Photo">` : ''}
                </div>
                <div class="post-actions">
                    <span class="like-btn ${isLiked ? 'liked' : ''}" onclick="likePost('${postId}')">
                        <i class="fa-${isLiked ? 'solid' : 'regular'} fa-heart"></i> ${d.likes || 0}
                    </span>
                </div>
            </div>
        `;
    });
});

// ЛОГИКА ЛАЙКОВ (Умная: 1 человек = 1 лайк)
function likePost(id) {
    const myNik = localStorage.getItem('social_nik');
    if(!my Nik) { alert("Сначала впиши свой ник в профиле!"); return; }

    const postRef = db.collection("posts").doc(id);

    db.runTransaction(async (transaction) => {
        const postDoc = await transaction.get(postRef);
        if (!postDoc.exists) return;

        const d = postDoc.data();
        const likedBy = d.likedBy || [];
        const currentLikes = d.likes || 0;

        if (likedBy.includes(myNik)) {
            // Если уже лайкнул — убираем лайк
            const newLikedBy = likedBy.filter(name => name !== myNik);
            transaction.update(postRef, {
                likes: Math.max(0, currentLikes - 1),
                likedBy: newLikedBy
            });
        } else {
            // Если еще не лайкнул — добавляем
            likedBy.push(myNik);
            transaction.update(postRef, {
                likes: currentLikes + 1,
                likedBy: likedBy
            });
        }
    });
}
